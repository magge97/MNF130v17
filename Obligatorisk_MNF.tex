\item [a)]
La $P(n)$ være påstanden $a_n = 2^{n-1}$. Vil vise ved induksjon at $P(n)$ er sann for alle heltall $n \geq 1$.
\begin{description}
\item[Basissteg]
Vi ser at $a_1 = 1 = 2^{1-1}$ og $a_2 = 1 + 1 = 2^{2-1}$, så $P(1)$ og $P(2)$ er sann. 
\item[Induksjonshypotese] Anta at $P(j)$ er sann for all heltall $j$ med $1 \leq j \leq k$ for $k \geq 2$, altså at $a_j = 2^{j-1}$.
\item[Induksjonssteg] Ønsker å vise at $P(k+1)$ er sann, altså at $a_{k+1} = 2^{(k+1)-1} = 2^k$. Vi ser at
\begin{align*}
a_{k+1} = 1 + \sum_{i=1}^k a_i = 1 + \sum_{i=1}^{k-1} a_i + a_k = a_k + a_k = 2a_k = 2 \cdot 2^{k-1} = 2^k.
\end{align*}
\end{description}

% 4b)
\item [b)]
La $P(n)$ være påstanden $f_n \leq \left(\frac{7}{4} \right)^n$. Vil vise ved induksjon at $P(n)$ er sann for alle heltall $n \geq 0$. 
\begin{description}
\item[Basissteg] Vi ser at $a_0 = 0 \leq \left(\frac{7}{4} \right)^0$ og $a_1 = 1 \leq \left(\frac{7}{4} \right)^1 $, så $P(0)$ og $P(1)$ er sann.
\item[Induksjonsshypotese] Anta at $P(j)$ er sann for alle heltall $j$ med $0 \leq j \leq k$ for $k \geq 1$, altså at $f_j \leq \left(\frac{7}{4} \right)^j$.
\item[Induksjonssteg] Ønsker å vise at $P(k+1)$ er sann, altså at $f_{k+1} \leq \left(\frac{7}{4} \right)^{k+1}$. Vi ser at 
\begin{align*}
f_{k+1} = f_k + f_{k-1} \leq \left(\frac{7}{4} \right)^{k} + \left(\frac{7}{4} \right)^{k-1} = \left(\frac{7}{4} \right)^{k+1} \left(\left(\frac{7}{4} \right)^{-1} + \left(\frac{7}{4} \right)^{-2} \right) < \left(\frac{7}{4} \right)^{k+1}.
\end{align*}
\end{description}
\end{itemize}

\section{}
\begin{itemize}

% 1a)
\item [a)] La $P(n)$ være påstanden $3^n < n!$. Vil vise ved induksjon at $P(n)$ er sann for alle heltall $n \geq 7$.
\begin{description}
\item[Basissteg] Vi ser at $3^7 < 7!$, altså at $P(1)$ er sann.
\item[Induksjonshypotese] Anta at $P(k)$ er sann for $k \geq 7$, altså at $3^k < k!$.
\item[Induksjonssteg] Ønsker å vise at $P(k+1)$ er sann, altså at $3^{k+1} < {(k+1)}!$. Vi ser at
\begin{align*}
3^{k+1} &= 3^k \cdot 3 < k! \cdot 3 < k! \cdot (k+1) = (k+1)!.
\end{align*}
\end{description}

% 1b)
\item [b)] La $P(n)$ være påstanden 
\begin{align*}
\sum_{i=0}^{n}(-7)^{i} = \frac{1-(-7)^{n+1}}{8}.
\end{align*} 
Vil vise ved induksjon at $P(n)$ er sann for alle heltall $n \geq 0$.
\begin{description}
\item[Basissteg] Vi ser at $\frac{1-(-7)^{0+1}}{8} = \frac{8}{8} = 1 = (-7)^0$, altså at $P(0)$ er sann.
\item[Induksjonshypotese] Anta at $P(k)$ er sann for $k \geq 0$, altså at 
\begin{align*}
\sum_{i=0}^{k}(-7)^{i} = \frac{1-(-7)^{k+1}}{8}.
\end{align*} 
\item[Induksjonssteg] Ønsker å vise at $P(k+1)$ er sann, altså at 
\begin{align*}
\sum_{i=0}^{k+1}(-7)^{i} = \frac{1-(-7)^{k+2}}{8}.
\end{align*}  
Vi ser at
\begin{align*}
\sum_{i=0}^{k+1}(-7)^{i} &= \sum_{i=0}^{k}(-7)^{i} + (-7)^{k+1} \\
&= \frac{1-(-7)^{k+1}}{8} + \frac{8\cdot(-7)^{k+1}}{8} \\
&= \frac{1+7\cdot(-7)^{k+1}}{8} \\
&= \frac{1-(-7)^{k+2}}{8}
\end{align*}
\end{description}

% 1b)
\item [c)] La $P(n)$ være påstanden $10^n \equiv 1(\mod 3)$. Vil vise ved induksjon at $P(n)$ er sann for alle heltall $n \geq 0$.
\begin{description}
\item[Basissteg] Vi ser at $3$ deler $1-1 = 0$, altså at $P(0)$ er sann.
\item[Induksjonshypotese] Anta at $P(k)$ er sann for $k \geq 0$, altså at $10^k \equiv 1(\mod 3)$.
\item[Induksjonssteg] Ønsker å vise at $P(k+1)$ er sann, altså at $10^{k+1} \equiv 1(\mod 3)$. Fra induksjonshypotesen har vi at $10^{k} \equiv 1(\mod 3)$ og $10 \equiv 1(\mod 3)$. Da følger det at \begin{align*}
10^k \cdot 10 \equiv 1\cdot1(\mod 3) \Longleftrightarrow 10^{k+1} \equiv 1(\mod 3).
\end{align*}
\end{description}
\end{itemize}
